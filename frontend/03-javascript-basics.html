➕ Operators

// ============================================ // ARITHMETIC OPERATORS - Math operations // ============================================ let a = 10; let b = 3; console.log(a + b); // 13 (Addition - jodna) console.log(a - b); // 7 (Subtraction - ghatana) console.log(a * b); // 30 (Multiplication - guna) console.log(a / b); // 3.333... (Division - bhag) console.log(a % b); // 1 (Modulus - remainder/baki) console.log(a ** b); // 1000 (Exponent - power) // Increment/Decrement let count = 5; count++; // 6 (count = count + 1) count--; // 5 (count = count - 1) // ============================================ // COMPARISON OPERATORS - Compare karna // ============================================ console.log(10 == "10"); // true (value same, type different) console.log(10 === "10"); // false (strict - value aur type dono) console.log(10 != "10"); // false console.log(10 !== "10"); // true (strict not equal) console.log(10 > 5); // true (greater than - bada) console.log(10 < 5); // false (less than - chota) console.log(10 >= 10); // true (greater or equal) console.log(10 <= 5); // false (less or equal) // ============================================ // LOGICAL OPERATORS - Logic lagana // ============================================ let age = 25; let hasLicense = true; // && (AND - dono true hone chahiye) console.log(age >= 18 && hasLicense); // true // || (OR - ek bhi true ho) console.log(age < 18 || hasLicense); // true // ! (NOT - opposite) console.log(!hasLicense); // false // ============================================ // STRING OPERATORS - String operations // ============================================ let firstName = "Sai"; let lastName = "Kumar"; let fullName = firstName + " " + lastName; // "Sai Kumar" // Template literals (modern way) let greeting = `Hello, ${firstName}! 🔀 Conditionals (If/Else)

// ============================================ // IF ELSE - Conditions check karna // ============================================ let age = 20; // Simple if if (age >= 18) { console.log("Adult hai"); } // If-else if (age >= 18) { console.log("Vote kar sakte ho"); } else { console.log("Abhi chote ho"); } // If-else if-else (multiple conditions) let marks = 85; if (marks >= 90) { console.log("Grade A+"); } else if (marks >= 80) { console.log("Grade A"); } else if (marks >= 70) { console.log("Grade B"); } else if (marks >= 60) { console.log("Grade C"); } else { console.log("Fail"); } // Nested if (if ke andar if) let isLoggedIn = true; let isPremium = true; if (isLoggedIn) { if (isPremium) { console.log("Premium content access"); } else { console.log("Free content access"); } } // ============================================ // TERNARY OPERATOR - Short if-else // ============================================ let canVote = (age >= 18) ? "Yes" : "No"; console.log(canVote); // "Yes" // ============================================ // SWITCH CASE - Multiple conditions // ============================================ let day = 3; let dayName; switch(day) { case 1: dayName = "Monday"; break; case 2: dayName = "Tuesday"; break; case 3: dayName = "Wednesday"; break; default: dayName = "Invalid day"; }

🔁 Loops (Repeat karna)

// ============================================ // FOR LOOP - Fixed number of times // ============================================ // 1 se 5 tak print for (let i = 1; i <= 5; i++) { console.log("Number: " + i); } // Array loop let fruits = ["Apple", "Banana", "Mango"]; for (let i = 0; i < fruits.length; i++) { console.log(fruits[i]); } // ============================================ // WHILE LOOP - Condition true tak chalta // ============================================ let count = 1; while (count <= 5) { console.log("Count: " + count); count++; } // ============================================ // DO WHILE - Pehle execute, phir check // ============================================ let num = 1; do { console.log("Number: " + num); num++; } while (num <= 5); // ============================================ // FOR...OF - Arrays ke liye modern way // ============================================ let colors = ["Red", "Green", "Blue"]; for (let color of colors) { console.log(color); } // ============================================ // FOR...IN - Objects ke liye // ============================================ let person = { name: "Sai", age: 25, city: "Mumbai" }; for (let key in person) { console.log(key + ": " + person[key]); } // ============================================ // ARRAY METHODS (Modern way) // ============================================ // forEach - Har item pe function run karo let numbers = [1, 2, 3, 4, 5]; numbers.forEach(function(num) { console.log(num * 2); }); // Arrow function (modern syntax) numbers.forEach(num => console.log(num * 2));

🔧 Functions

// ============================================ // FUNCTION DECLARATION - Traditional way // ============================================ // Simple function (no parameters) function sayHello() { console.log("Hello!"); } sayHello(); // Call karna // Function with parameters function greet(name) { console.log("Hello, " + name); } greet("Sai"); // Hello, Sai // Function with return value function add(a, b) { return a + b; // Value return karna } let sum = add(5, 3); // 8 // Default parameters function multiply(a, b = 2) { return a * b; } console.log(multiply(5)); // 10 (5 * 2) console.log(multiply(5, 3)); // 15 (5 * 3) // ============================================ // FUNCTION EXPRESSION - Variable me function // ============================================ const subtract = function(a, b) { return a - b; }; console.log(subtract(10, 5)); // 5 // ============================================ // ARROW FUNCTIONS - Modern short syntax // ============================================ // Traditional const addOld = function(a, b) { return a + b; }; // Arrow (short) const addNew = (a, b) => { return a + b; }; // Arrow (even shorter - implicit return) const addShorter = (a, b) => a + b; // Single parameter (parentheses optional) const square = num => num * num; // ============================================ // CALLBACK FUNCTIONS - Function as parameter // ============================================ function processNumbers(a, b, callback) { return callback(a, b); } let result1 = processNumbers(5, 3, (x, y) => x + y); // 8 let result2 = processNumbers(5, 3, (x, y) => x * y); // 15 // ============================================ // IMMEDIATELY INVOKED FUNCTION (IIFE) // ============================================ (function() { console.log("I run immediately!"); })();

📚 Arrays

// ============================================ // ARRAY BASICS // ============================================ // Create array let fruits = ["Apple", "Banana", "Mango"]; let numbers = [1, 2, 3, 4, 5]; // Access elements (0 se start) console.log(fruits[0]); // "Apple" console.log(fruits[1]); // "Banana" // Length console.log(fruits.length); // 3 // Change element fruits[1] = "Orange"; // ============================================ // ARRAY METHODS - Common operations // ============================================ let items = ["A", "B", "C"]; // Add elements items.push("D"); // End me add ["A","B","C","D"] items.unshift("Z"); // Start me add ["Z","A","B","C","D"] // Remove elements items.pop(); // End se remove ["Z","A","B","C"] items.shift(); // Start se remove ["A","B","C"] // Find element let index = items.indexOf("B"); // 1 let exists = items.includes("B"); // true // Remove from middle items.splice(1, 1); // Index 1 se 1 element remove // Get part of array let sliced = items.slice(0, 2); // Pehle 2 elements // Join array to string let str = items.join(", "); // "A, B, C" // Reverse array items.reverse(); // ["C", "B", "A"] // Sort array let nums = [3, 1, 4, 1, 5, 9]; nums.sort(); // [1, 1, 3, 4, 5, 9] // ============================================ // MODERN ARRAY METHODS // ============================================ let numbers = [1, 2, 3, 4, 5]; // map - Transform each element let doubled = numbers.map(num => num * 2); // [2, 4, 6, 8, 10] // filter - Keep only matching elements let evens = numbers.filter(num => num % 2 === 0); // [2, 4] // reduce - Combine into single value let sum = numbers.reduce((total, num) => total + num, 0); // 15 // find - Find first match let found = numbers.find(num => num > 3); // 4 // some - Check if any match let hasEven = numbers.some(num => num % 2 === 0); // true // every - Check if all match let allPositive = numbers.every(num => num > 0); // true

📦 Objects

// ============================================ // OBJECT BASICS - Key-value pairs // ============================================ // Create object let person = { name: "Sai", age: 25, city: "Mumbai", isStudent: false, hobbies: ["coding", "music", "sports"] }; // Access properties console.log(person.name); // Dot notation console.log(person["age"]); // Bracket notation // Add new property person.phone = "9876543210"; // Modify property person.age = 26; // Delete property delete person.isStudent; // ============================================ // OBJECT METHODS - Functions inside object // ============================================ let user = { name: "Sai", age: 25, // Method (function inside object) greet: function() { console.log("Hello, " + this.name); }, // Modern shorthand sayAge() { console.log("I am " + this.age); } }; user.greet(); // "Hello, Sai" user.sayAge(); // "I am 25" // ============================================ // OBJECT OPERATIONS // ============================================ // Get all keys let keys = Object.keys(person); // ["name", "age", "city", "hobbies"] // Get all values let values = Object.values(person); // ["Sai", 25, "Mumbai", [...]] // Get key-value pairs let entries = Object.entries(person); // [["name", "Sai"], ["age", 25], ...] // Check if property exists console.log("name" in person); // true // Clone object (shallow copy) let personCopy = {...person}; // Spread operator // Merge objects let details = {salary: 50000, dept: "IT"}; let combined = {...person, ...details}; // ============================================ // NESTED OBJECTS - Objects ke andar objects // ============================================ let student = { name: "Rahul", address: { city: "Mumbai", pincode: 400001 }, marks: { math: 90, english: 85 } }; console.log(student.address.city); // "Mumbai" console.log(student.marks.math); // 90

✅ JavaScript Summary

📌 Must Know Concepts:

  • Variables: let, const (data store karne ke liye)
  • Data Types: String, Number, Boolean, Array, Object
  • Operators: +, -, *, /, ==, ===, &&, ||
  • Conditionals: if, else, else if, switch
  • Loops: for, while, forEach
  • Functions: function keyword, arrow functions
  • Arrays: map, filter, reduce, forEach
  • Objects: Key-value pairs, methods
  • DOM: getElementById, querySelector, innerHTML
  • Events: addEventListener (MOST IMPORTANT!)

⚡ addEventListener Syntax:

element.addEventListener('click', function() { // Code yaha likho }); // Arrow function se element.addEventListener('click', () => { // Code yaha });
